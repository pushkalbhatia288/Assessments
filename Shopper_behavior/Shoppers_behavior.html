np.random.seed(1)
# Initialising the ANN
classifier = Sequential()

# Adding the input layer and the first hidden layer
classifier.add(Dense(units = 20, activation = 'relu', input_dim = 24))

# Adding the second hidden layer

# Included L2 Regularizer to prevent overfitting
classifier.add(Dense(units = 12, kernel_regularizer=l2(0.01), bias_regularizer=l2(0.01), activation = 'relu'))

classifier.add(Dense(units = 6, kernel_regularizer=l2(0.01), bias_regularizer=l2(0.01), activation = 'relu'))

# Adding the output layer
classifier.add(Dense(units = 1, activation = 'sigmoid'))

# Compiling the ANN
classifier.compile(optimizer = 'adam', loss = 'binary_crossentropy', metrics = ['accuracy'])

# Fitting the ANN to the Training set
